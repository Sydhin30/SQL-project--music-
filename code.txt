
  
  Question 1 - most popular genre 
  SELECT G.NAME GENRE,
  	SUM(IL.UNITPRICE)
  FROM CUSTOMER C
  JOIN INVOICE I ON I.CUSTOMERID = C.CUSTOMERID
  JOIN INVOICELINE IL ON IL.INVOICEID = I.INVOICEID
  JOIN TRACK T ON T.TRACKID = IL.TRACKID
  JOIN GENRE G ON G.GENREID = T.GENREID
  GROUP BY 1
  ORDER BY 2 DESC
  
  Question 2 - most popular R&B alumb
  SELECT a.Title, COUNT(t.TrackId) AS TrackCount
FROM Album a
JOIN Track t ON a.AlbumId = t.AlbumId
JOIN Genre g ON t.GenreId = g.GenreId
WHERE g.Name = 'R&B/Soul'
GROUP BY a.Title
ORDER BY TrackCount DESC;


question 3 - most sales per state 
SELECT i.BillingState, SUM(i.Total) AS TotalSales
FROM Invoice i
JOIN Customer c ON i.CustomerId = c.CustomerId
WHERE c.Country = 'USA'
GROUP BY i.BillingState
ORDER BY TotalSales DESC;

 question 4 - What is the total revenue generated by each employee?
 SELECT e.FirstName || ' ' || e.LastName AS EmployeeName, SUM(i.Total) AS TotalRevenue
FROM Employee e
JOIN Customer c ON e.EmployeeId = c.SupportRepId
JOIN Invoice i ON c.CustomerId = i.CustomerId
GROUP BY EmployeeName
ORDER BY TotalRevenue DESC;
